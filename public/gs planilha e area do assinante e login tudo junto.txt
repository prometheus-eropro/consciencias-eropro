function doPost(e) {
  let dados;
  try {
    dados = JSON.parse(e.postData.contents);
  } catch (erro) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: "erro", acesso: false, mensagem: "Requisição inválida" }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  const login = (dados.login || "").trim();
  const senha = dados.senha || "";

  const ss = SpreadsheetApp.openById("1RyOi80PamsOKEomBfW42fGYSP9FBFLYMB4-giQcCK6c"); // Seu ID
  const sheetUsers = ss.getSheetByName("ACESSOS_CONSC");
  const sheetConsc = ss.getSheetByName("ConsciêncIAs");
  const sheetLogs = ss.getSheetByName("LOGS"); // se quiser log

  const dadosUsuarios = sheetUsers.getDataRange().getValues();

  let usuarioEncontrado = null;

  for (let i = 1; i < dadosUsuarios.length; i++) {
    const registro = dadosUsuarios[i];
    const emailOuCelular = registro[0] ? registro[0].toString().trim() : "";
    if (emailOuCelular.toLowerCase() === login.toLowerCase()) {
      usuarioEncontrado = registro;
      break;
    }
  }

  let resposta = { status: "", acesso: false };
  const dataHoraAtual = new Date();

  if (!usuarioEncontrado) {
    resposta.status = "erro";
    resposta.mensagem = "Usuário não encontrado.";
    if (sheetLogs) sheetLogs.appendRow([dataHoraAtual, login, "falha - usuário não encontrado"]);
  } else {
    const senhaCorreta = usuarioEncontrado[1] || "";
    if (senhaCorreta !== senha) {
      resposta.status = "erro";
      resposta.mensagem = "Senha incorreta.";
      if (sheetLogs) sheetLogs.appendRow([dataHoraAtual, login, "falha - senha incorreta"]);
    } else {
      const ativo = (usuarioEncontrado[2] || "").toLowerCase();
      if (ativo !== "sim") {
        resposta.status = "erro";
        resposta.mensagem = "Usuário inativo.";
        if (sheetLogs) sheetLogs.appendRow([dataHoraAtual, login, "falha - usuário inativo"]);
      } else {
        // Sucesso
        resposta.status = "ok";
        resposta.acesso = true;
        resposta.nome_cliente = usuarioEncontrado[3] || "";
        resposta.data_ativacao = usuarioEncontrado[5] || "";
        resposta.nome_consciencia = usuarioEncontrado[4] || "";

        // Buscar detalhes da consciência na aba "ConsciêncIAs"
        const nomeConsc = (usuarioEncontrado[4] || "").toString().trim();
        const dadosConsc = sheetConsc.getDataRange().getValues();
        let encontrada = dadosConsc.find((linha, idx) => idx > 0 && linha[0]?.toString().trim() === nomeConsc);

        if (encontrada) {
          resposta.consciencia_detalhes = {
            nome: encontrada[0] || "",
            resumo: encontrada[1] || "",
            descricao: encontrada[2] || "",
            como_usar: encontrada[3] || "",
            prompt: encontrada[4] || "",
            link_demo: encontrada[5] || "",
            link_premium: encontrada[6] || "",
            valor: encontrada[7] || "",
            promocao: encontrada[8] || ""
          };
        } else {
          resposta.mensagem = "Consciência não encontrada.";
        }

        if (sheetLogs) sheetLogs.appendRow([dataHoraAtual, login, "sucesso"]);
      }
    }
  }

  return ContentService
    .createTextOutput(JSON.stringify(resposta))
    .setMimeType(ContentService.MimeType.JSON);
